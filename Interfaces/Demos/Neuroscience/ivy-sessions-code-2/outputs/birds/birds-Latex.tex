\subsection{birds (long)}

\begin{questionUnit}{Is it true that ' Birds are funny. '?}
	\Context{}{
		\lEntry{Birds}{animal}
		\lEntry{funny}{}
	}
	\Answers{}{
		\Statement{Yes.}{
			\Context{}{
				\Statement{They make me laugh.}{
					\Context{}{
						\lEntry{laugh}{}
						\Statement{paper 1}{}
					}}
				\Statement{They bathe in puddles.}{
					\Context{}{
						\Statement{paper 2}{}
					}}
				\Statement{I sit at on a bench and watch birds.}{}
			}}
		\Statement{No.}{
			\Context{}{
				\Statement{They can be dirty.}{}
			}}
		\Statement{The question doesn't make sense or is irrelevant.}{
			\Context{}{
				\Statement{Squirrels might be even more funny.}{}
			}}
	}
\end{questionUnit}
